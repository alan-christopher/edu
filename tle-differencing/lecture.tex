\subsection{Mean Element Theories}

\begin{frame}{Perturbation: Non-Sphericity}
  \[ \ddot{R} = \ddot{R}_\text{2B} + \alert{\ddot{R}_\text{NS}} \]
  \begin{itemize}
  \item $\ddot{R}_\text{2B}$ uses point-mass equations
    \begin{itemize}
    \item Works for points
    \item Works for spheres (shell theorem)
    \end{itemize}
  \item Earth is neither
    \begin{itemize}
    \item Centrifugal forces (order kilometers)
    \item Tidal forces (order meters)
    \end{itemize}
  \item Notable for causing nodal precession.
  \end{itemize}
\end{frame}

\begin{frame}{Perturbation: Drag}
  \[ \ddot{R} = \ddot{R}_\text{2B} + \ddot{R}_\text{NS} + \alert{\ddot{R}_\text{D}} \]

  \begin{itemize}
  \item Drag equation: $F_\text{D} = \frac{1}{2}\rho v^2 C_\text{D} A$
  \item Scales by
    \begin{itemize}
    \item Object shape and orientation ($C_\text{D}, A$).
    \item Square of object velocity $v^2$.
    \item Atmospheric density $\rho$, drops exponentially with altitude.
    \end{itemize}
  \item LEO objects (altitude $< 2000$ km) are low and fast
    \begin{itemize}
    \item experience non-negligible drag
    \end{itemize}
  \item Bonus: non-periodic and non-conservative.
  \end{itemize}
\end{frame}


\begin{frame}{Perturbation: et cetera}
  We've only scratched the surface.
  \begin{itemize}
  \item Third bodies (esp. sun, moon)
  \item Tidal forces (solid and not)
  \item Solar radiation pressure (esp. HAMR)
  \item Knocke rediffusion
  \item Relativistic corrections
    \begin{itemize}
    \item Schwarzschild
    \item Lense-Thirring
    \item De Sitter
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Differencing}
  \begin{center}
    \includegraphics[scale=0.32]{images/fig1.png}
  \end{center}

  \begin{enumerate}
    \only<1>{\item[1.] Select an RSO's TLE history to examine.}
    \only<2>{\item[2.] Assume residuals $\epsilon$ are consistently distributed
      over an \emph{Analysis Window}.}
    \only<3>{\item[3.] Choose a \emph{Propagation Window} over which SGP4 should
      be ``good enough''.}
    \only<4>{\item[4.] Compute $\epsilon(\Delta t)$ samples by computing pairwise differences against reference TLEs.}
    \only<5>{\item[5a.] Fit function $\hat{\epsilon}(\Delta t)$ to gathered samples.}
    \only<5>{\item[5b.] Compute covariance statistics from gathered samples.}
  \end{enumerate}
\end{frame}

\begin{frame}{Reference State}
  Using TLEs at epoch as reference states limits our pool of residual
  samples. What if we merged all our TLEs together into a reference state, a la
  sample means?

  \pause
  
  \[ x_\text{ref} = \sum{\bar{w_i} x_i} \]

  \pause

  $x_i$ is less reliable the further from $t_\text{ref}$ it is. Choose $w_i$
  accordingly
  
  \[ w_i = \frac{1}{\text{age}^2} \]
  
  \pause

  Why inverse square? No theoretical justification provided.
\end{frame}

\begin{frame}{Outlier Handling: Rejection Multipliers}
  TLEs are crap sometimes -- need to manage outliers. \pause Iterative procedure:

  \begin{enumerate}
  \item Calculate sample mean, sample deviation $\epsilon_i$s
  \item Discard if too many deviations away from mean. Common admissibility
    range $r_\text{max}$ is somewhere between 3 and 6.
  \item If discard, back to top.
  \end{enumerate}\pause

  Commonly effective, but $r_\text{max}$ is a nuisance hyper-parameter.
\end{frame}

\begin{frame}{Outlier Handling: Weighting}
  What if we downweighted dubious samples instead of tossing them entirely? \pause

  \begin{enumerate}
  \item Perform least squares fit.
  \item Compute new weights from fit residuals (e.g., via bi-square weight function).
  \item Repeat until convergence or loss of patience.
  \end{enumerate}\pause
  
  WLSQ and Bi-square weight function are topics for another day, but in summary:
  \begin{enumerate}
  \item WLSQ minimizes a weighted L2 norm.
  \item Bi-square weight gives more weight to well-modeled obs and less to
    poorly modeled obs.
  \end{enumerate}  
\end{frame}
